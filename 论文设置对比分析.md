```python
class SurgicalVideoDataset(Dataset):
    def __init__(self,
                 spatial_size=(224, 224),
                 horizontal_flip_prob=0.5,  # 水平翻转概率
                 rotation_degrees=[-15, 0, 15],  # 旋转角度
                 color_jitter=0.1,  # 颜色抖动
                 crop_scale=0.9,  # 随定剪裁比例
                 normalize=True,
                 ...):
        self.horizontal_flip_prob = horizontal_flip_prob
        self.rotation_degrees = rotation_degrees
        self.color_jitter = color_jitter
        self.crop_scale = crop_scale
        self.normalize = normalize
```

2. **在 __getitem__ 中应用数据增强**:
```python
def __getitem__(self, idx):
    frames = self._load_frames(idx)

    # 水平翻转
    if self.horizontal_flip_prob > 0 and random.random() < self.horizontal_flip_prob:
        frames = torch.flip(frames, dims=[-2])

    # 随机旋转
    if self.rotation_degrees and random.random() < 0.5:
        angle = random.choice(self.rotation_degrees)
        frames = torch.rot90(frames, angle, dims=[2])

    # 裁色抖动
    if self.color_jitter > 0:
        jitter = torch.randn_like(frames) * self.color_jitter
        frames = torch.clamp(frames + jitter, 0, 1)

    # 随机裁剪
    if self.crop_scale < 1.0 and random.random() < 0.5:
        # 实现中心随机裁剪
        pass

    # 归一化
    if self.normalize:
        frames = frames * 2.0 - 1.0  # [-1, 1]

    return frames
```
