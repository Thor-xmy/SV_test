```python
def __getitem__(self, idx):
    frames = self._load_frames(idx)
    masks = self._load_masks(idx) if self.use_mask else None

    # === 数据增强 (仅训练时) ===
    if self.is_train:
        # 水平翻转
        if hasattr(self, 'horizontal_flip_prob') and self.horizontal_flip_prob > 0:
            if random.random() < self.horizontal_flip_prob:
                frames = torch.flip(frames, dims=[-2])  # 水平翻转
                if masks is not None:
                    masks = torch.flip(masks, dims=[-1])

        # 随机旋转 (90度倍数)
        if hasattr(self, 'enable_rotation') and self.enable_rotation:
            if random.random() < 0.5:
                k = random.choice([1, 2, 3])  # 90, 180, 270度
                frames = torch.rot90(frames, k=k, dims=[-2, -1])
                if masks is not None:
                    masks = torch.rot90(masks, k=k, dims=[0, 1])

    # 归一化
    if self.normalize:
        frames = (frames - 0.5) * 2.0  # [-1, 1]

    return {'frames': frames, 'masks': masks, ...}
```

## 🛠️ 完整修复方案

### 步骤1: 修改学习率设置

**文件**: `train.py`

在 `build_optimizer` 函数中添加分学习率支持：

```python
def build_optimizer(model, config):
    """Build optimizer with differential learning rates."""
    # 论文要求:
    # - backbone: 1e-4
    # - static module: 1e-3

    backbone_lr = config.get('backbone_lr', 1e-4)
    static_lr = config.get('static_lr', 1e-3)

    # 分组参数
    backbone_params = []
    static_params = []

    for name, param in model.named_parameters():
        if 'static_extractor' in name or 'dynamic_extractor' in name:
            backbone_params.append(param)
        else:
            static_params.append(param)

    # 构建参数组
    param_groups = []
    if backbone_params:
        param_groups.append({'params': backbone_params, 'lr': backbone_lr})
    if static_params:
        param_groups.append({'params': static_params, 'lr': static_lr})

    optimizer = optim.Adam(
        param_groups,
        weight_decay=config.get('weight_decay', 0.0)  # 论文要求0
    )

    print(f"Optimizer: Adam")
    print(f"  Backbone LR: {backbone_lr}")
    print(f"  Static Module LR: {static_lr}")
    print(f"  Weight Decay: {config.get('weight_decay', 0.0)}")

    return optimizer
```
