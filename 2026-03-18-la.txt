  def train_epoch(model, dataloader, optimizer, device, epoch, config):
      for batch in dataloader:
          # 1. 准备数据
          video = batch['video'].to(device)   # (B, 3, T, H, W)
          masks = batch['masks'].to(device)   # (B, T, H, W)
          score_gt = batch['score'].to(device)   # (B,) 在[0,1]

          # 2. 前向传播
          score_pred = model(video, masks)  # (B, 1) 在[0,1]

          # 3. 计算损失
          loss = F.mse_loss(score_pred.squeeze(-1), score_gt)

          # 4. 反向传播
          optimizer.zero_grad()
          loss.backward()
          optimizer.step()

● 详细解释：实际clip数与期望不符的处理逻辑

  第202-214行代码：

  # 5. Handle case where actual num_clips differs from expected_clips
  actual_input_dim = temporal_features.shape[1]          # 实际输入维度
  expected_input_dim = self.expected_clips * self.total_clip_dim  # 期望输入维度

  if actual_input_dim != expected_input_dim:
      # 需要调整特征维度以匹配回归器的输入维度

          padding = torch.zeros(B, expected_input_dim - actual_input_dim,
  device=video.device)
          temporal_features = torch.cat([temporal_features, padding], dim=1)

  ---
  为什么要这段代码？

  问题根源：
  - fusion_regressor在__init__中创建，需要固定输入维度
  - 我们用expected_clips（如10）来初始化
  - 但实际视频可能有不同的clip数（如9或11）

  例子：

  ┌──────────┬─────────────┬────────────────┬─────────────────┐
  │ 视频帧数 │ 实际clips数 │ expected_clips │    处理方式     │
  ├──────────┼─────────────┼────────────────┼─────────────────┤
  │ 100帧    │ 10          │ 10             │ ✅ 无需处理     │
  ├──────────┼─────────────┼────────────────┼─────────────────┤
  │ 90帧     │ 9           │ 10             │ 🟡 Padding填充0 │
  ├──────────┼─────────────┼────────────────┼─────────────────┤
  │ 50帧     │ 5           │ 10             │ 🟡 Padding填充0 │
  ├──────────┼─────────────┼────────────────┼─────────────────┤
  │ 110帧    │ 11          │ 10             │ 🟡 Truncate截断 │
  └──────────┴─────────────┴────────────────┴─────────────────┘

  ---
  情况A：实际 < 期望（Padding填充0）

  例子：90帧视频
  视频帧数：90
  clip_length = 16, clip_stride = 10
  实际clips数 = (90-16)//10 + 1 = 9

  静态特征：fused_per_clip = (B, 9, 1536)
  展平后：temporal_features = (B, 9 * 1536) = (B, 13824)

  期望输入：expected_input_dim = 10 * 1536 = 15360

  因为 13824 < 15360：
  padding = torch.zeros(B, 15360 - 13824) = (B, 1536)

  temporal_features = cat([(B, 13824), (B, 1536)], dim=1)
                  = (B, 15360)  ← 匹配回归器输入！

  可视化：
  实际clips特征：[clip1, clip2, ..., clip9]  → (B, 13824)
  Padding：       [0, 0, ..., 0]       → (B, 1536)
  合并后：         [clip1, clip2, ..., clip9, 0, ..., 0]  → (B, 15360)
                                               ↑
                                               缺少的clip用0填充

  ---
  情况B：实际 > 期望（Truncate截断）

  例子：110帧视频
  视频帧数：110
  clip_length = 16, clip_stride = 10
  实际clips数 = (110-16)//10 + 1 = 11

  静态特征：fused_per_clip = (B, 11, 1536)
  展平后：temporal_features = (B, 11 * 1536) = (B, 16896)

  期望输入：expected_input_dim = 10 * 1536 = 15360

  因为 16896 > 15360：
  temporal_features = temporal_features[:, :15360]
                             = (B, 15360)  ← 截断到回归器输入！

  可视化：
  实际clips特征：[clip1, clip2, ..., clip9, clip10, clip11]  → (B, 16896)
  截断后：         [clip1, clip2, ..., clip9, clip10]          → (B, 15360)
                                              ↑
                                              丢弃多余的clip

  ---
  优缺点分析

  ┌──────────┬────────────────────────────┬─────────────────────────────────┐
  │ 处理方式 │            优点            │              缺点               │
  ├──────────┼────────────────────────────┼─────────────────────────────────┤
  │ Padding  │ • 保留所有clip信息•        │ • 填充的0是噪声•                │

  │ Truncate │ • 无噪声• 维度精确匹配     │ • 丢失末尾clip信息•             │
  │          │                            │ 可能遗漏重要操作                │
  └──────────┴────────────────────────────┴─────────────────────────────────┘
